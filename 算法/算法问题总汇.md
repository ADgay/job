##### 简单方法：

- 选择排序

  - 运行时间与输入无关，需要N^2/2次比较与N次交换．
  - 数据移动最少．

  ```c++
  //Here is your code
  vector<int> sorted(vector<int> A) {
      int N = A.size();
      for(int i = 0;i<N;++i){
          int min_val = A[i];
          for(int j = i;j<N;++j){
              if(A[j]<min_val){
                  min_val = A[j];
                  exch(&A[i], &A[j]);
              }
          }
      }
      return A;
  
  }
  //end
  ```

- 插入排序

  - 排序速度与数据分布有关，当数据部分有序时速度较快
  - 需要较多的交换次数

```c++
//Here is your code
vector<int> sorted(vector<int> A) {
    int N = A.size();
    for(int i = 1;i<N;++i){
        int j = i;
        while(A[j]<A[j-1]&&j>=1){
            exch(&A[j], &A[j-1]);
            --j;
        }
    }
    return A;
    }

//end
```

- 希尔排序
  - 唯一无法准确描述其对乱序数组的性能特征．
  - 一次移动更远的距离，来提高运算速度.
  - 在数据量并不是特别大时，速度仅比高级排序慢一两倍，而且代码简单，五额外空间．

```c++
//Here is your code
vector<int> sorted(vector<int> A) {
    int N = A.size();
    int h;
    while(h<N/3) h = h*3+1;
    while(h>=1){
        for(int i = h;i<N;++i){
            int j = i;
            while(A[j]<A[j-h]&&j>=h){
                exch(&A[j], &A[j-h]);
                j -= h;
            }
        }
        h /= 3;
    }
    return A;
    }
//end
```

```c++
void sort_core(vector<int>* Ap,vector<int>* Auxp,  int a, int b){
    if(b<=a) return;
    sort_core(Ap, Auxp,  a, a+(b-a)/2);
    sort_core(Ap, Auxp,  a+(b-a)/2+1, b);
    merge(Ap, Auxp,  a, a+(b-a)/2 , b);
}
void merge(vector<int>* Ap, vector<int>* Auxp, int a, int b, int c){
    int l1 = a;
    int m1 = b+1;

    for(int i = a;i<=c;++i){
        (*Auxp)[i] = (*Ap)[i];
    }
//下面的for循环写的挺好，因为肯定要判断c-a+1次．
    for(int k = a;k<=c;++k){
        if(l1>b) (*Ap)[k] = (*Auxp)[m1++];
        else if(m1>c) (*Ap)[k] = (*Auxp)[l1++];
        else if((*Auxp)[l1] < (*Auxp)[m1]) (*Ap)[k] = (*Auxp)[l1++];
        else (*Ap)[k] = (*Auxp)[m1++];
    }

}
```

- 快速排序
  - 在递归过程中创建数组是巨大的开销．
  - 可以在数组比较小的时候直接调用插入排序．
  - 双指针中间卡住的时候要与设定的中值交换位置
  - 三取样切分，使用三个数的中位数作为切分元素
  - 熵最优排序：Dijkstra，三向切分排序，增加一个指针存放与切分元素相等的值，该方法在有较多重复元素的排序中速度较快．（算法４中图解做的很好），也就是荷兰国旗问题的解．

```C++
//Here is your code

vector<int> sorted(vector<int> A){
    int N = A.size();
    sort_core(&A, 0, N-1);
    return A;
}

void sort_core(vector<int>* A, int a, int b){
    if(b<=a) return;
    // print_self(A);
    int l1 = a;
    int m2 = b;
    int val = (*A)[b];
    while(true){
        while((*A)[l1]<=val && l1<b) ++l1;
        while((*A)[m2]>=val && m2>a) --m2;
        if(l1>=m2) break;
        exch(&((*A)[l1]), &((*A)[m2]));
    }
    exch(&((*A)[l1]), &((*A)[b]));//这一步很关键
    
    sort_core(A, a, l1-1);
    sort_core(A, l1+1, b);
}

//end
```

```C++
void swim(vector<int>* A, int k){
    while(k>1){
        if((*A)[k]<(*A)[k/2]){
            exch(&((*A)[k]), &((*A)[k/2]));
            k /= 2;
        }
    }
}

void sink(vector<int>* A, int k, int MAX_n){
    int N = MAX_n;
    if(MAX_n==-2) N = (*A).size()-1;
    while(k*2<=N){
        int j = k*2;
        if(j<N && (*A)[j]>(*A)[j+1]) ++j;
        if((*A)[j]<(*A)[k]){
            exch(&((*A)[k]), &((*A)[j]));
            k = j;
        }else break;
        }
}
```
